388      public void calculateWeanDates() { 
389          LocalDate dob = getDOB(); 
390          est34Date = dob.plusWeeks(8).toString(); 
391          est12Date = dob.plusWeeks(10).toString(); 
392          est14Date = dob.plusWeeks(11).toString(); 
393          estWaterDate = dob.plusWeeks(12).toString(); 
394          estMoveDate = dob.plusWeeks(13).toString(); 
395      }
